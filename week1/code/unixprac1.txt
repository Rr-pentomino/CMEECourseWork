# Counting lines for every file in the folder using relative path
wc -l  ../data/fasta/*

# Print everything starting from the second line
tail -n +2 ../data/fasta/E.coli.fasta

# Count the sequence length of this genome (removing the headline and newlines)
tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\n' | wc -c

# Count the matches of a particular sequence, “ATGC” 
tail -n +2 ../data/fasta/E.coli.fasta |  tr -d '\r\n' | grep -o 'ATGC' | wc -l

# Compute the AT/GC ratio (without removing uncertain bases)
AT=$(tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\r\n' | tr -cd 'AT' | wc -c); GC=$(tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\r\n' |tr -cd 'GC' | wc -c);  echo "scale=4; $AT/$GC" | bc -l
